Aplanador PDF

Analiza las características de un PDF y, solo si tiene cifrado o restricciones, genera una copia plana sin contraseñas ni limitaciones. Si el PDF está limpio, no hace nada.

Ejemplo A: limpiar PDF al enviar un formulario

  
Intercepta el submit. Al pulsar
  Enviar el PDF se aplana y se sustituye en el FormData antes del envío.
  El servidor recibe siempre un PDF limpio.



<!-- HTML del formulario -->
<form id="miForm" action="/api/upload" method="POST" enctype="multipart/form-data">
  <label>Nombre: <input type="text" name="nombre"></label>
  <label>PDF: <input type="file" id="pdfInput" name="documento" accept=".pdf"></label>
  <button type="submit">Enviar</button>
</form>

<!-- Dependencias -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.min.js"></script>
<script>pdfjsLib.GlobalWorkerOptions.workerSrc =
  "https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js";</script>
<script src="https://cdnjs.cloudflare.com/ajax/libs/pdf-lib/1.17.1/pdf-lib.min.js"></script>
<script src="pdf-flatten.min.js"></script>  <!-- o incluir PdfFlatten inline -->

<script>
// Se intercepta el submit del formulario.
// flatten() renderiza cada pagina a imagen y genera un PDF nuevo
// sin cifrado, contrasenas, restricciones ni metadatos sensibles.
// Devuelve un objeto File con el nombre original + sufijo _pl.pdf

document.getElementById("miForm").addEventListener("submit", async function(e) {
  e.preventDefault();

  var file = document.getElementById("pdfInput").files[0];
  if (!file) return;

  // Aplanar: convierte el PDF a imagenes dentro de un PDF limpio
  var clean = await PdfFlatten.flatten(file);
  // clean.name = "documento_pl.pdf", clean.size = bytes del nuevo PDF

  // Sustituir el fichero original por el limpio en el FormData
  var fd = new FormData(this);
  fd.set("documento", clean, clean.name);

  // Enviar al servidor (el servidor recibe el PDF ya limpio)
  await fetch(this.action, { method: "POST", body: fd });
});
</script>



  
  
Ejemplo B: analizar primero y aplanar solo si hace falta

  
Usa analyze() para inspeccionar
  el PDF. Solo lo aplana si detecta cifrado. Muestra un resumen del análisis y,
  si hay contraseña de apertura, la pide al usuario.



<input type="file" id="pdfInput" accept=".pdf">

<script>
document.getElementById("pdfInput").addEventListener("change", async function() {
  var file = this.files[0];
  if (!file) return;

  // ---- PASO 1: analizar el PDF ----
  // analyze(file, password?) devuelve un objeto con:
  //   .needsPassword    - true si requiere contrasena de apertura
  //   .wrongPassword    - true si la contrasena proporcionada es incorrecta
  //   .fileName         - nombre del fichero
  //   .fileSize         - tamano en bytes
  //   .pdfVersion       - "1.7", "1.6", etc.
  //   .numPages         - numero de paginas
  //   .creator          - aplicacion que creo el PDF
  //   .producer         - motor que genero el PDF
  //   .creationDate     - fecha de creacion (formato PDF)
  //   .modDate          - fecha de modificacion
  //   .encryption       - objeto con info de cifrado:
  //       .encrypted    - boolean
  //       .algorithm    - "AES-256", "RC4-128", "Ninguno", etc.
  //       .keyLength    - longitud de clave en bits
  //       .V, .R        - version y revision del cifrado PDF
  //       .P            - valor raw de permisos
  //       .permissions  - objeto desglosado:
  //           .print, .printHQ, .modify, .copy,
  //           .annotate, .fillForms, .accessibility, .assemble
  //   .pages[]          - array con { number, widthPt, heightPt }

  var info = await PdfFlatten.analyze(file);

  // Si requiere contrasena, pedirla
  if (info.needsPassword) {
    var pw = prompt("Contrasena de apertura:");
    if (!pw) return;
    info = await PdfFlatten.analyze(file, pw);
    if (info.needsPassword) { alert("Contrasena incorrecta"); return; }
  }

  console.log("Paginas: " + info.numPages);
  console.log("Cifrado: " + info.encryption.algorithm);

  // ---- PASO 2: aplanar solo si esta cifrado ----
  if (info.encryption.encrypted) {
    // flatten(file, password?, opciones?)
    //   opciones.scale      - escala de renderizado (defecto: 2, mayor = mas calidad)
    //   opciones.quality    - calidad JPEG 0..1 (defecto: 0.92)
    //   opciones.onProgress - callback(pagina, total) para mostrar progreso
    var clean = await PdfFlatten.flatten(file, pw, {
      scale: 2,
      quality: 0.92,
      onProgress: function(pag, total) {
        console.log("Pagina " + pag + " de " + total);
      }
    });
    // clean es un File: clean.name = "archivo_pl.pdf"
    console.log("Resultado: " + clean.name + " (" + clean.size + " bytes)");
  } else {
    console.log("Sin cifrado, no necesita aplanado");
  }
});
</script>



  
  
Ejemplo C: todo en un paso con process()

  
process() ejecuta
  analyze() + flatten() en una sola llamada y descarga
  el resultado automáticamente.



<input type="file" id="pdfInput" accept=".pdf">

<script>
document.getElementById("pdfInput").addEventListener("change", async function() {
  var file = this.files[0];
  if (!file) return;

  // process(file, password?, opciones?)
  // Ejecuta analyze + flatten en un solo paso.
  // Devuelve un objeto con:
  //   .analysis  - mismo resultado que analyze() (ver Ejemplo B)
  //   .flatFile  - File con el PDF limpio (null si necesita contrasena)
  var result = await PdfFlatten.process(file);

  // Si necesita contrasena, reintentar
  if (result.analysis.needsPassword) {
    var pw = prompt("Contrasena de apertura:");
    result = await PdfFlatten.process(file, pw);
  }

  // Mostrar info del analisis
  var a = result.analysis;
  console.log("Cifrado: " + a.encryption.algorithm);
  console.log("Paginas: " + a.numPages);

  // Descargar el PDF limpio
  if (result.flatFile) {
    var url = URL.createObjectURL(result.flatFile);
    var link = document.createElement("a");
    link.href = url;
    link.download = result.flatFile.name;  // nombre_pl.pdf
    link.click();
    URL.revokeObjectURL(url);
  }
});
</script>
